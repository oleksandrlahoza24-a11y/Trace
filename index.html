'#define UINT8_MODE\n' : '') + ` precision highp float; varying vec2 vUV; uniform sampler2D uWater; uniform vec2 uTexel; uniform float uTime; uniform float uAspect; float hash(vec2 p) { return fract(sin(dot(p, vec2(127.1,311.7))) * 43758.5453); } float vnoise(vec2 p) { vec2 i = floor(p), f = p - i; f = f*f*(3.0 - 2.0*f); return mix(mix(hash(i), hash(i+vec2(1,0)), f.x), mix(hash(i+vec2(0,1)),hash(i+vec2(1,1)),f.x), f.y); } float fbm(vec2 p) { float v=0.0, a=0.5; for(int i=0;i<5;i++){ v+=vnoise(p)*a; p*=2.1; a*=0.5; } return v; } vec3 getSky(vec2 dir) { float t = clamp(dir.y * 1.4, 0.0, 1.0); vec3 sky = mix(vec3(0.3, 0.6, 0.9), vec3(0.02, 0.1, 0.3), t); vec2 sunD = normalize(vec2(0.3, 0.7)); float sd = dot(normalize(dir), sunD); sky += vec3(1.0, 0.9, 0.7) * pow(max(sd, 0.0), 300.0) * 5.0; return sky; } // IMPROVED SEABED: Sand ripples and sediment vec3 getSeabed(vec2 uv) { vec2 p = uv * 6.0; float ripples = sin(p.x + fbm(p) * 3.0) * 0.5 + 0.5; ripples *= sin(p.y * 0.5 + fbm(p + 1.5) * 2.0) * 0.5 + 0.5; vec3 sand = mix(vec3(0.76, 0.7, 0.5), vec3(0.9, 0.85, 0.7), ripples); float detail = fbm(uv * 40.0); sand = mix(sand, sand * 0.9, detail * 0.4); // Ambient occlusion/darker deep spots float darken = smoothstep(0.4, 0.9, fbm(uv * 3.0)); sand = mix(sand, sand * 0.6, darken); return sand; } #ifdef UINT8_MODE float H(vec2 uv) { vec4 s = texture2D(uWater, uv); return (s.r + s.g / 255.0) * 2.0 - 1.0; } #else float H(vec2 uv) { return texture2D(uWater, uv).r * 2.0 - 1.0; } #endif float GGX(float NdotH, float rough) { float a2 = rough*rough*rough*rough; float d = NdotH*NdotH*(a2-1.0)+1.0; return a2 / (3.14159*d*d + 1e-5); } void main() { vec2 uv = vUV; vec2 tx = uTexel * 1.5; float h = H(uv); float hL = H(uv - vec2(tx.x, 0.0)); float hR = H(uv + vec2(tx.x, 0.0)); float hU = H(uv + vec2(0.0, tx.y)); float hD = H(uv - vec2(0.0, tx.y)); // Calculate Normal vec3 normal = normalize(vec3(hL - hR, hD - hU, 0.15)); vec3 V = vec3(0.0, 0.0, 1.0); vec3 L = normalize(vec3(0.5, 0.5, 1.0)); // FRESNEL float fresnel = pow(1.0 - max(dot(normal, V), 0.0), 5.0); fresnel = mix(0.05, 1.0, fresnel); // REFRACTION & SEABED vec2 refOff = normal.xy * 0.04; vec3 seabed = getSeabed(uv + refOff); // CAUSTICS float lap = hL + hR + hU + hD - 4.0 * h; float caustics = pow(max(0.0, -lap * 12.0), 1.5) * 2.5; // Fake moving light rays on seabed float lightRays = pow(vnoise(uv * 4.0 + uTime * 0.1), 3.0) * 0.5; seabed *= (1.0 + caustics + lightRays); // Beer-Lambert Water Tint vec3 waterTint = mix(vec3(0.0, 0.4, 0.5), vec3(0.0, 0.05, 0.1), clamp(1.0 - h*3.0, 0.0, 1.0)); seabed = mix(seabed, waterTint, 0.4); // REFLECTION vec3 R = reflect(-L, normal); vec3 sky = getSky(R.xy); // FINAL COMBINATION vec3 col = mix(seabed, sky, fresnel); // Add specular highlight float spec = pow(max(dot(reflect(-L, normal), V), 0.0), 64.0); col += vec3(1.0, 0.9, 0.8) * spec * 0.5; // Tone mapping col = col / (col + vec3(1.0)) * 1.1; gl_FragColor = vec4(col, 1.0); }`;
